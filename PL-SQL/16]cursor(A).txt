a) The bank manager has decided to activate all those accounts which were previously marked as 
inactive for performing no transaction in last 365 days. Write a PL/SQ block (using implicit 
cursor) to update the status of account, display an approximate message based on the no. of rows 
affected by the update. (Use of %FOUND, %NOTFOUND, %ROWCOUNT)

mysql> CREATE TRIGGER update_account_status BEFORE INSERT ON transactions
     FOR EACH ROW
     BEGIN
       UPDATE accounts
       SET account_status = 'Active'
       WHERE account_id = NEW.account_id;
     END;

mysql> DELIMITER //
mysql> CREATE PROCEDURE activate_inactive_accounts()
     BEGIN
       DECLARE v_account_id INT;
       DECLARE v_last_transaction_date DATE;
       DECLARE v_rows_updated INT DEFAULT 0;
       DECLARE done INT DEFAULT 0;
       DECLARE cur CURSOR FOR
         SELECT account_id, last_transaction_date
         FROM accounts
         WHERE account_status = 'Inactive'
         AND last_transaction_date < DATE_SUB(NOW(), INTERVAL 365 DAY);
       DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = 1;
    
       OPEN cur;
    
       read_loop: LOOP
         FETCH cur INTO v_account_id, v_last_transaction_date;
         IF done = 1 THEN
          LEAVE read_loop;
         END IF;
    
        UPDATE accounts
         SET account_status = 'Active'
         WHERE account_id = v_account_id;
    
         SET v_rows_updated = v_rows_updated + 1;
       END LOOP;
    
       CLOSE cur;
    
       IF v_rows_updated = 0 THEN
         SELECT 'No inactive accounts were updated.';
       ELSE
         SELECT CONCAT(v_rows_updated, ' inactive accounts were updated.');
       END IF;
     END;
     //

mysql> DELIMITER ;
mysql> CALL activate_inactive_accounts();
